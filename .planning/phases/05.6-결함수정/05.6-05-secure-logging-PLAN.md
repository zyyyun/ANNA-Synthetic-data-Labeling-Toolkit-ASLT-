HMAC wire format (from CONTEXT specifics):
`logline|prev_hmac=abc|hmac=def`
- prev_hmac=GENESIS for first line
- Day rollover: new file first line uses previous file's last HMAC as prev_hmac (경계 무결성)

Key file spec:
- Path: `%LOCALAPPDATA%\ANNA\ASLT\.hmac-key`
- 32 bytes random (RandomNumberGenerator.GetBytes)
- Read-only for current user (FileSecurity + FileSystemAccessRule 또는 최소 File.SetAttributes Hidden)
- 파일 없으면 자동 생성 (머신 최초 실행)

Serilog extension points:
- `Serilog.Core.ILogEventSink` — 커스텀 Sink 인터페이스

```csharp
using System;
using System.IO;
using System.Security.Cryptography;
using Serilog;

namespace ASLTv1.Services
{
    /// <summary>
    /// D-17b: HMAC 체인용 비밀키 관리 (%LOCALAPPDATA%\ANNA\ASLT\.hmac-key, 32 bytes).
    /// </summary>
    public static class HmacKeyProvider
    {
        private static readonly string KeyDir = Path.Combine(
            Environment.GetFolderPath(Environment.SpecialFolder.LocalApplicationData),
            "ANNA", "ASLT");
        private static readonly string KeyPath = Path.Combine(KeyDir, ".hmac-key");

        public static byte[] GetOrCreateKey()
        {
            Directory.CreateDirectory(KeyDir);

            if (File.Exists(KeyPath))
            {
                byte[] existing = File.ReadAllBytes(KeyPath);
                if (existing.Length == 32) return existing;
                Log.Warning("[HMAC 키] 기존 키 파일 크기 비정상 ({Length}) — 재생성", existing.Length);
            }

            byte[] key = new byte[32];
            using (var rng = RandomNumberGenerator.Create())
                rng.GetBytes(key);

            File.WriteAllBytes(KeyPath, key);
            try { File.SetAttributes(KeyPath, FileAttributes.Hidden); } catch { /* best-effort */ }
            Log.Information("[AUDIT] HMAC 키 생성: {Path}", KeyPath);
            return key;
        }
    }
}
```

```csharp
using System;
using System.IO;
using System.Security.Cryptography;
using System.Text;
using Serilog.Core;
using Serilog.Events;
using Serilog.Formatting;

namespace ASLTv1.Services
{
    /// <summary>
    /// D-17b: Serilog ILogEventSink 구현 — 각 로그 라인에 `|prev_hmac=...|hmac=...` 부착.
    /// </summary>
    public class HmacChainSink : ILogEventSink
    {
        private readonly string _pathTemplate;   // 예: %LOCALAPPDATA%\ANNA\ASLT\logs\AOLT-{date}.log
        private readonly ITextFormatter _formatter;
        private readonly byte[] _key;
        private readonly object _lock = new object();
        private string _lastHmac = null;
        private string _lastFilePath = null;

        public HmacChainSink(string pathTemplate, ITextFormatter formatter, byte[] key)
        {
            _pathTemplate = pathTemplate ?? throw new ArgumentNullException(nameof(pathTemplate));
            _formatter = formatter ?? throw new ArgumentNullException(nameof(formatter));
            _key = key ?? throw new ArgumentNullException(nameof(key));
            Directory.CreateDirectory(Path.GetDirectoryName(pathTemplate));
        }

        public void Emit(LogEvent logEvent)
        {
            lock (_lock)
            {
                string logDir = Path.GetDirectoryName(_pathTemplate);
                string baseName = Path.GetFileNameWithoutExtension(_pathTemplate);  // "AOLT-"
                string ext = Path.GetExtension(_pathTemplate);                       // ".log"
                string dateStr = logEvent.Timestamp.ToString("yyyy-MM-dd");
                string filePath = Path.Combine(logDir, $"{baseName}{dateStr}{ext}");

                // 일자 경계 넘어간 경우 이전 파일의 마지막 hmac 읽기
                if (_lastFilePath != filePath)
                {
                    _lastHmac = ReadLastHmac(filePath);
                    _lastFilePath = filePath;
                }

                // 베이스 메시지 직렬화
                using var writer = new StringWriter();
                _formatter.Format(logEvent, writer);
                string baseLine = writer.ToString().TrimEnd('\r', '\n');

                string prevHmac = _lastHmac ?? "GENESIS";
                string payload = baseLine + "|prev_hmac=" + prevHmac;
                string hmacHex = ComputeHmacHex(payload);

                string finalLine = baseLine + "|prev_hmac=" + prevHmac + "|hmac=" + hmacHex + Environment.NewLine;

                File.AppendAllText(filePath, finalLine, Encoding.UTF8);
                _lastHmac = hmacHex;
            }
        }

        private string ReadLastHmac(string filePath)
        {
            if (!File.Exists(filePath)) return null;
            // 마지막 줄 읽기 (큰 파일 대비 끝에서 역순 스캔 가능하지만, 일별 분할이므로 전체 읽어도 수 MB 수준)
            string last = null;
            foreach (var line in File.ReadLines(filePath))
            {
                if (!string.IsNullOrWhiteSpace(line)) last = line;
            }
            if (string.IsNullOrEmpty(last)) return null;

            const string marker = "|hmac=";
            int idx = last.LastIndexOf(marker, StringComparison.Ordinal);
            if (idx < 0) return null;
            return last.Substring(idx + marker.Length).TrimEnd();
        }

        private string ComputeHmacHex(string data)
        {
            using var h = new HMACSHA256(_key);
            byte[] hash = h.ComputeHash(Encoding.UTF8.GetBytes(data));
            var sb = new StringBuilder(hash.Length * 2);
            foreach (byte b in hash) sb.Append(b.ToString("x2"));
            return sb.ToString();
        }
    }
}
```

```csharp
using System;
using System.IO;
using System.Security.Cryptography;
using System.Text;

namespace ASLTv1.Services
{
    public static class LogIntegrityVerifier
    {
        public class Result
        {
            public bool IsValid { get; set; }
            public int? TamperedLine { get; set; }
            public string Reason { get; set; }
        }

        public static Result VerifyLogFile(string logFilePath, byte[] key, string initialPrevHmac = "GENESIS")
        {
            if (!File.Exists(logFilePath))
                return new Result { IsValid = false, Reason = "파일이 존재하지 않습니다: " + logFilePath };

            string prev = initialPrevHmac;
            int lineNum = 0;
            using var reader = new StreamReader(logFilePath, Encoding.UTF8);
            string line;
            while ((line = reader.ReadLine()) != null)
            {
                lineNum++;
                if (string.IsNullOrWhiteSpace(line)) continue;

                int hmacIdx = line.LastIndexOf("|hmac=", StringComparison.Ordinal);
                int prevIdx = line.LastIndexOf("|prev_hmac=", StringComparison.Ordinal);
                if (hmacIdx < 0 || prevIdx < 0 || prevIdx > hmacIdx)
                    return new Result { IsValid = false, TamperedLine = lineNum, Reason = "HMAC 마커 누락" };

                string baseLine = line.Substring(0, prevIdx);
                string prevHmac = line.Substring(prevIdx + "|prev_hmac=".Length, hmacIdx - (prevIdx + "|prev_hmac=".Length));
                string actualHmac = line.Substring(hmacIdx + "|hmac=".Length);

                if (!prevHmac.Equals(prev, StringComparison.Ordinal))
                    return new Result { IsValid = false, TamperedLine = lineNum, Reason = "prev_hmac 체인 불일치" };

                string expected = ComputeHmacHex(key, baseLine + "|prev_hmac=" + prevHmac);
                if (!expected.Equals(actualHmac, StringComparison.Ordinal))
                    return new Result { IsValid = false, TamperedLine = lineNum, Reason = "HMAC 재계산 불일치" };

                prev = actualHmac;
            }

            return new Result { IsValid = true };
        }

        private static string ComputeHmacHex(byte[] key, string data)
        {
            using var h = new HMACSHA256(key);
            byte[] hash = h.ComputeHash(Encoding.UTF8.GetBytes(data));
            var sb = new StringBuilder(hash.Length * 2);
            foreach (byte b in hash) sb.Append(b.ToString("x2"));
            return sb.ToString();
        }
    }
}
```

```csharp
private static void CleanupOldLogs(string logDir, int retainDays)
{
    try
    {
        DateTime cutoff = DateTime.Now.AddDays(-retainDays);
        foreach (var file in Directory.GetFiles(logDir, "AOLT-*.log"))
        {
            var info = new FileInfo(file);
            if (info.LastWriteTime < cutoff)
            {
                File.Delete(file);
                // NOTE: 로그 자체 미동작 상태이므로 Console.WriteLine 또는 생략
            }
        }
    }
    catch (Exception ex)

        Log.Warning(ex, "[오래된 로그 정리 실패]");
    }
}
```
